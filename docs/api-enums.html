枚举类型 (Enums)

WeAutoCommon 库中定义的所有枚举类型。

ChatType

聊天类型，用于标识会话的类型。

public enum ChatType
{
    好友,        // 与好友单聊
    群聊,        // 群聊
    公众号,      // 公众号
    订阅号,      // 订阅号
    服务通知,    // 服务通知
    腾讯新闻,    // 腾讯新闻
    微信团队,    // 微信团队
    其他         // 其他类型
}

MessageType

消息类型，定义聊天消息的类型。

public enum MessageType
{
    文字,        图片,  语音,  视频,  文件,
    聊天记录,    位置,  个人名片,  视频号,
    视频号直播,  红包,  微信转账,  小程序,
    链接,        表情,  时间,  引用,  笔记,
    拍一拍,      撤回消息,  其他
}

MessageSourceType

消息来源类型，标识消息的来源。

public enum MessageSourceType
{
    系统消息,
    自己发送消息,
    好友消息,
    其他消息
}

NavigationType

导航类型，定义微信主窗口的导航类型。

public enum NavigationType
{
    聊天,
    通讯录,
    收藏,
    聊天文件,
    朋友圈,
    视频号,
    看一看,
    搜一搜,
    小程序面板,
    手机,
    设置及其他
}

ConversationType

会话类型，用于标识会话的类型。

public enum ConversationType
{
    好友,
    群聊,
    公众号,
    订阅号,
    服务通知,
    腾讯新闻,
    微信团队,
    其他
}

ChatBoxToolBarType

聊天框工具栏类型，定义聊天框工具栏中的按钮类型。

public enum ChatBoxToolBarType
{
    表情,
    发送文件,
    截图,
    聊天记录,
    直播,
    语音聊天,
    视频聊天
}

ChatContentType

聊天内容类型，定义聊天内容的类型。

public enum ChatContentType
{
    MainWindow,  // 主窗口
    SubWindow    // 子窗口
}

ActionType

操作类型，定义各种自动化操作的类型。

public enum ActionType
{
    打开子窗口,
    发送消息,
    发送文件,
    发送表情
}

CopyType

复制类型，定义复制操作的类型。

public enum CopyType
{
    文本,
    图片,
    文件
}

使用示例

// 判断聊天类型
var chatType = chatContent.ChatType;
if (chatType == ChatType.群聊)
{
    Console.WriteLine("这是一个群聊");
}

// 判断消息类型
var messageType = messageBubble.MessageType;
if (messageType == MessageType.文字)
{
    Console.WriteLine("这是一条文字消息");
}

// 切换导航栏
client.NavigationSwitch(NavigationType.朋友圈);
